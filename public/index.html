async function loadProducts() { const container = document.getElementById('products-container'); try { // Запрос идет прямо к твоему серверу const response = await fetch('/products'); const products = await response.json(); if (!products || products.length === 0) { container.innerHTML = '
Магазин пуст
'; return; } container.innerHTML = ''; products.forEach(product => { const card = document.createElement('div'); card.className = 'product-card'; card.innerHTML = `

${product.title}

${product.description || ''}

${product.price} ₽
`; container.appendChild(card); }); } catch (err) { console.error("Ошибка во фронтенде:", err); if (container) { container.innerHTML = '
Не удалось загрузить товары
'; } } } // Запускаем функцию при загрузке страницы document.addEventListener('DOMContentLoaded', loadProducts);
